# helper functions to generate archetype shapes
supernova_curve <- function(t) {
  peak_day <- 5
  peak <- dnorm(t, mean = peak_day, sd = 3)
  tail <- dnorm(t, mean = peak_day + 7, sd = 15)
  (peak + 0.4 * tail) / max(peak + 0.4 * tail)
}

slowburn_curve <- function(t) {
  growth <- plogis((t - 40) / 6)
  decay <- exp(-(t - 40) / 80)
  curve <- growth * decay
  curve <- ifelse(t < 5, 0, curve)
  curve / max(curve)
}

undying_curve <- function(t) {
  base <- 0.5 + 0.1 * sin(2 * pi * t / 7) + 0.15 * sin(2 * pi * t / 30)
  trend <- 0.3 + 0.3 * exp(-(t - 30) / 300)
  curve <- pmax(base * trend, 0)
  curve / max(curve)
}

dormant_revival_curve <- function(t) {
  base <- rep(0.02, length(t))
  revival <- dnorm(t, mean = 80, sd = 5)
  curve <- base + 2 * revival
  curve / max(curve)
}
